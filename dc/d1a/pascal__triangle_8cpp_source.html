36int **pascal_triangle(int **arr, int n) {
37 for (int i = 0; i < n; ++i) {
38 for (int j = n - i - 1; j < n + i; ++j) {
39 if (j == n - i - 1 || j == n + i - 1)
40 arr[i][j] = 1; // The edge of the Pascal triangle goes in 1
41 else
42 arr[i][j] = arr[i - 1][j - 1] + arr[i - 1][j + 1];
43 }
44 }
45
46 return arr;
47}
