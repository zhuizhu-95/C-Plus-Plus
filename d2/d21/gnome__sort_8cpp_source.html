34void gnomeSort(T *arr, int size) {
35 // few easy cases
36 if (size <= 1) {
37 return;
38 }
39
40 int index = 0; // initialize some variables.
41 while (index < size) {
42 // check for swap
43 if ((index == 0) || (arr[index] >= arr[index - 1])) {
44 index++;
45 } else {
46 std::swap(arr[index], arr[index - 1]); // swap
47 index--;
48 }
49 }
