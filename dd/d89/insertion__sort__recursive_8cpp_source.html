50template <typename T>
51void insertionSort(T *arr, int n) {
52 for (int i = 1; i < n; i++) {
53 T temp = arr[i];
54 int j = i - 1;
55 while (j >= 0 && temp < arr[j]) {
56 arr[j + 1] = arr[j];
57 j--;
58 }
59 arr[j + 1] = temp;
60 }
